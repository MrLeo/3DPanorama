theme : 'ta',

//设置回调句柄
}
});
</script>

三、月份选择器：

开始时间：2012-06-14
结束时间：2012-07-10

示例代码：需额外引入日期的js和css

<script type="text/javascript">
monthPicker.create('month_picker', {
trigger : 'month_trigger',
autoCommit : true,
callback : function(obj){
$("#monthContainer").html('开始时间 : ' + obj.startDate + '
结束时间 : ' + obj.endDate);
}
});
</script>
